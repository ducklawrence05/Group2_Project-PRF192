Tính chu vi và diện tích hình vuông
Source
#include <stdio.h>
    
int main() {
  float canh, chuVi, dienTich;

  printf("Nhập chiều dài cạnh của hình vuông: ");
  scanf("%f", &canh);

  //kiểm tra cạnh có phải số dương không
  if (canh > 0) {
    chuVi = 4 * canh;
    dienTich = canh * canh;

    printf("Chu vi hình vuông: %.2f\n", chuVi);
    printf("Diện tích hình vuông: %.2f\n", dienTich);
  } else {
    printf("Chiều dài cạnh phải là số dương.\n");
  }

  return 0;
}
Input
Nhập chiều dài cạnh của hình vuông: 3.5
Output
Chu vi hình vuông: 14.00
Diện tích hình vuông: 12.25
