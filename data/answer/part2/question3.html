Tính chu vi và diện tích hình chữ nhật
Source
#include <stdio.h>

int main() {
  float dai, rong, chuVi, dienTich;

  printf("Nhập chiều dài của hình chữ nhật: ");
  scanf("%f", &dai);
  printf("Nhập chiều rộng của hình chữ nhật: ");
  scanf("%f", &rong);

  //kiểm tra dài và rộng là số dương
  if (dai > 0 && rong > 0) {
    chuVi = 2 * (dai + rong);
    dienTich = dai * rong;

    printf("Chu vi hình chữ nhật: %.2f\n", chuVi);
    printf("Diện tích hình chữ nhật: %.2f\n", dienTich);
  } else {
    printf("Chiều dài và chiều rộng phải là các số dương.\n");
  }

  return 0;
}
Input
Nhập chiều dài của hình chữ nhật: 7.5
Nhập chiều rộng của hình chữ nhật: 4.2
Output
Chu vi hình chữ nhật: 23.40
Diện tích hình chữ nhật: 31.50
