Tính chu vi và diện tích tam giác từ 3 cạnh nhập vào 
Source
#include <stdio.h>
#include <math.h>

int main() {
  float a, b, c, p, chuVi, dienTich;

  printf("Nhập độ dài ba cạnh tam giác: ");
  scanf("%f %f %f", &a, &b, &c);

  // Kiểm tra điều kiện tam giác
  if (a + b > c && a + c > b && b + c > a) {
    chuVi = a + b + c;
    p = chuVi / 2; // p là nửa chu vi
    // Tính diện tích theo công thức Heron
    dienTich = sqrt(p * (p - a) * (p - b) * (p - c));

    printf("Chu vi tam giác: %.2f\n", chuVi);
    printf("Diện tích tam giác: %.2f\n", dienTich);
  } else {
    printf("Ba cạnh không tạo thành tam giác hợp lệ.\n");
  }

  return 0;
}
Input
Nhập độ dài ba cạnh tam giác: 3 4 5
Output
Chu vi tam giác: 12.00
Diện tích tam giác: 6.00
