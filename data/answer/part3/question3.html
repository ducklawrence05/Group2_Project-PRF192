Kiểm tra mảng các số nguyên là chẵn hay lẻ
Source
#include <stdio.h>
#define MAX_SIZE 100 // Đặt giới hạn kích thước mảng

int main() {
  int n;
  int arr[MAX_SIZE]; // Khai báo mảng tĩnh với kích thước tối đa là MAX_SIZE

  printf("Nhập số lượng phần tử (tối đa %d): ", MAX_SIZE);
  scanf("%d", &n);

  // Kiểm tra xem n có vượt quá kích thước mảng tối đa không
  if (n > MAX_SIZE) {
      printf("Số lượng phần tử quá lớn! Chỉ có thể nhập tối đa %d phần tử.\n", MAX_SIZE);
      return 1; // Thoát nếu n vượt quá MAX_SIZE
  }

  printf("Nhập các số nguyên:\n");
  for (int i = 0; i < n; i++) {
      scanf("%d", &arr[i]);
  }

  printf("Kết quả kiểm tra:\n");
  for (int i = 0; i < n; i++) {
      if (arr[i] % 2 == 0) {
          printf("%d là số chẵn.\n", arr[i]);
      } else {
          printf("%d là số lẻ.\n", arr[i]);
      }
  }  
  return 0;
}
Input
Nhập số lượng phần tử (tối đa 100): 5
Nhập các số nguyên:
4
7
2
11
8
Output
Kết quả kiểm tra:
4 là số chẵn.
7 là số lẻ.
2 là số chẵn.
11 là số lẻ.
8 là số chẵn.
